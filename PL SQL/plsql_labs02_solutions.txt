     create or replace function get_new_salary(p_job_id varchar2, p_salary number) return number
     is
        v_new_sal number(7,2);
     begin
        if p_job_id = 'ST_MAN' then
           v_new_sal := p_salary*1.1;
        elsif p_job_id = 'ST_CLERK' then
           v_new_sal := p_salary*1.05;
        else
           v_new_sal := p_salary*1.15;
        end if;
        
        return v_new_sal;
      end;

   wyliczenie podwyżki ma zostać zrealizowane wg następujących reguł:
   a. jeśli przekazane stanowisko to ST_MAN (manager) to funkcja powinna zwracać wartość 10
   b. jeśli przekazane stanowisko to IT_PROG to funkcja powinna zwracać wartość 20
   c. w pozostałych przypadkach funkcja powinna zwracać wartość 5

create or replace function get_raise_percent(p_job_id varchar2) return number
is
begin
	   if p_job_id = 'ST_MAN' then
	   		return 10;
	   elsif p_job_id = 'IT_PROG' then
	   		return 20;
	   end if;
	   return 5;
end;

   i zwracać wartość typu liczbowego reprezentującą ROK przejścia na emeryturę wg wzoru: 
   1. rok zatrudnienia uzyskany z daty zatrudnienia + 35 lat pracy dla wszystkich stanowisk oprócz SA_REP i ST_PROG
   2. rok zatrudnienia uzyskany z daty zatrudnienia + 25 lat pracy dla ST_PROG
   3. rok zatrudnienia uzyskany z daty zatrudnienia + 40 lat pracy dla SA_REP

create or replace function get_year_of_retirement(p_hire_date date, p_job_id varchar2) return number
is
	v_hire_year number(4) := to_number(to_char(p_job_id,'YYYY'));
begin
	if p_job_id = 'SA_REP' then
		return v_hire_year+40;
	elsif p_job_id = 'ST_PROG' then
		return v_hire_year+25;
	end if;
	return v_hire_year+35;
end;
